# Card

Displays a card with header, content, and footer.

## Usage

```jsx
import * as React from 'react'
import {
  Card,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from '@openui-org/react'

function App() {
  return (
    <Card className="w-96">
      <CardHeader>
        <img
          src="https://avatars.githubusercontent.com/u/116913262?v=4"
          className="w-full h-72 rounded-sm"
        />
      </CardHeader>
      <CardContent>
        <div className="text-xl font-medium pb-4 flex justify-between w-full">
          <CardTitle className="w-fill">Sebastián García</CardTitle>
          <span>Github</span>
        </div>
        <span className="text-sm font-extralight">
          Lorem ipsum dolor sit amet consectetur adipisicing elit. Explicabo,
          voluptatum?
        </span>
      </CardContent>
      <CardFooter className="flex justify-between">
        <Button className="w-full">Follow</Button>
      </CardFooter>
    </Card>
  )
}
```
