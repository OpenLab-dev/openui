# Aspect ratio

Displays content within a desired ratio.

## Usage

```jsx
import { AspectRatio } from '@openui-org/react/components'

export default function AspectRatioExample() {
  return (
    <div className="w-96">
      <AspectRatio ratio={16 / 9}>
        <img
          src="https://avatars.githubusercontent.com/u/116913262?v=4"
          alt="Image"
          className="rounded-md object-cover"
        />
      </AspectRatio>
    </div>
  )
}
```
