import {
  Button,
  Input,
  Label,
  TabsContent,
  Tabs,
  TabsList,
  TabsTrigger,
} from '@openui-org/react/components'
import Preview from 'components/preview'

# Tabs

Un conjunto de secciones de contenido en capas conocidas como paneles de pestañas que se muestran de una en una.

## Usar

```jsx
import * as React from 'react'
import {
  Button,
  Input,
  Label,
  TabsContent,
  Tabs,
  TabsList,
  TabsTrigger,
} from '@openui-org/react/components'

function TabsExample() {
  return (
    <Tabs defaultValue="Inicio de sesión" className="w-[400px]">
      <TabsList>
        <TabsTrigger value="Inicio de sesión">Inicio de sesión</TabsTrigger>
        <TabsTrigger value="Registrarse">Registrarse</TabsTrigger>
      </TabsList>
      <TabsContent value="Inicio de sesión">
        <form className="flex flex-col gap-4 text-start">
          <div>
            <Label>Correo</Label>
            <Input type="email" placeholder="jondoe@gmail.com" />
          </div>
          <div>
            <Label>Contraseña</Label>
            <Input type="password" placeholder="*******" />
          </div>
          <Button className="w-full">Iniciar sesión</Button>
        </form>
      </TabsContent>
      <TabsContent value="Registrarse">
        <form className="flex flex-col gap-2 text-start">
          <div>
            <Label>Nombre</Label>
            <Input type="text" placeholder="John Abraham" />
          </div>
          <div>
            <Label>Correo</Label>
            <Input type="email" placeholder="jondoe@gmail.com" />
          </div>
          <div>
            <Label>Contraseña</Label>
            <Input type="password" placeholder="*******" />
          </div>
          <div>
            <Label>Numero de telefono</Label>
            <Input type="text" placeholder="Ingrese us numero de telefono" />
          </div>
          <Button className="w-full mt-5">Registrarse</Button>
        </form>
      </TabsContent>
    </Tabs>
  )
}
```
